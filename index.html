function drawChart(price, bull, bear, date, json) { const chart = echarts.init(document.getElementById("chart")); const lastPrice = Number(json.furtherData.hsilast); const bullHeavy = computeHeavyZones(bull); const bearHeavy = computeHeavyZones(bear); const heavyZones = price.map((label, i) => { if (bull[i] >= bullHeavy) return "bull"; if (bear[i] >= bearHeavy) return "bear"; return ""; }); const option = { title: { text: `Bull / Bear Distribution — ${date}`, left: "center" }, tooltip: { trigger: "axis", axisPointer: { type: "shadow" }, formatter: function (params) { const i = params[0].dataIndex; const label = price[i]; const b = bull[i]; const r = bear[i]; const total = b + r; const heavy = heavyZones[i] ? `Heavy Zone (${heavyZones[i]})` : ""; const last = label.includes(lastPrice.toFixed(0)) ? "← Last Price" : ""; return ` ${label} ${last}
Bull: ${b}
Bear: ${r}
Total: ${total}
${heavy} `; } }, legend: { data: ["Bull", "Bear"], top: 30 }, grid: { left: 120, right: 40, top: 80, bottom: 40 }, xAxis: { type: "value", name: "Volume" }, yAxis: { type: "category", data: price, inverse: true }, series: [ { name: "Bull", type: "bar", stack: "total", data: bull, itemStyle: { color: "#ff66b3" } }, { name: "Bear", type: "bar", stack: "total", data: bear, itemStyle: { color: "#6666ff" } } ] }; chart.setOption(option); }
